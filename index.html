def pergunta_booleano(pergunta): while True: resposta = input(pergunta + " (sim/não): ").strip().lower() if resposta in ['sim', 's']: return True elif resposta in ['não', 'nao', 'n']: return False else: print("Por favor, responda com 'sim' ou 'não'.") print("Vamos pensar juntos sobre sua decisão.\n") quer_ir = pergunta_booleano("Você tem vontade de ir?") ficar_resolve = pergunta_booleano("Ficar resolve o problema?") print("\nEstou pensando...") if quer_ir and not ficar_resolve: print("Conclusão: Ir é a melhor escolha!") else: print("Conclusão: Melhor ficar mesmo.")
